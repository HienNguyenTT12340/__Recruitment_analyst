1.UV Đạt từng Level = 
VAR CurrentLvl = SELECTEDVALUE('Bảng mã tuyển dụng'[Mã])
VAR CurrentRank = SELECTEDVALUE('Bảng mã tuyển dụng'[Rank])

RETURN
IF(
    ISBLANK(CurrentLvl) || ISBLANK(CurrentRank), 
    BLANK(),
    VAR SelectedStartDate = MIN('Table_date'[Date])
    VAR SelectedEndDate = MAX('Table_date'[Date])
    
    RETURN
    SUMX(
        FILTER(
            ALLSELECTED('Bảng mã tuyển dụng'), -- Giữ lại filter từ Slicer lên bảng mã
            'Bảng mã tuyển dụng'[Rank] >= CurrentRank && NOT ISBLANK('Bảng mã tuyển dụng'[Mã])
        ),
        VAR LvlName = 'Bảng mã tuyển dụng'[Mã]
        VAR LvlRank = 'Bảng mã tuyển dụng'[Rank]
        
        RETURN
        CALCULATE(
            COUNT('HR_Recruitment tracker'[Id]),
            -- Thay ALL bằng KEEPFILTERS để bảo toàn các filter từ Slicer khác (Bộ phận, Team...)
            KEEPFILTERS('HR_Recruitment tracker'[Mã (L0-L5)] = LvlName),
            FILTER(
                ALLSELECTED('HR_Recruitment tracker'), -- ALLSELECTED giúp nhận diện filter từ Slicer Bộ phận
                (
                    ('HR_Recruitment tracker'[Created chuẩn] >= SelectedStartDate && 'HR_Recruitment tracker'[Created chuẩn] <= SelectedEndDate) ||
                    ('HR_Recruitment tracker'[Modified chuẩn] >= SelectedStartDate && 'HR_Recruitment tracker'[Modified chuẩn] <= SelectedEndDate) ||
                    ('HR_Recruitment tracker'[Created chuẩn] >= SelectedStartDate && 'HR_Recruitment tracker'[Modified chuẩn] <= SelectedEndDate)
                ) &&
                (
                    (LvlRank = CurrentRank && 'HR_Recruitment tracker'[Result] = "Pass") || 
                    (LvlRank > CurrentRank)
                )
            )
        )
    )
)

2.UV Reject = 
VAR CurrentLvl = SELECTEDVALUE('Bảng mã tuyển dụng'[Mã])

RETURN
IF(
    ISBLANK(CurrentLvl), 
    BLANK(),
    VAR SelectedStartDate = MIN('Table_date'[Date])
    VAR SelectedEndDate = MAX('Table_date'[Date])
    
    RETURN
    CALCULATE(
        COUNT('HR_Recruitment tracker'[Id]),
        'HR_Recruitment tracker'[Result] = "Reject",
        FILTER(
            ALLSELECTED('HR_Recruitment tracker'), -- Thay ALL bằng ALLSELECTED
            'HR_Recruitment tracker'[Mã (L0-L5)] = CurrentLvl && (
                ('HR_Recruitment tracker'[Created chuẩn] >= SelectedStartDate && 'HR_Recruitment tracker'[Created chuẩn] <= SelectedEndDate) ||
                ('HR_Recruitment tracker'[Modified chuẩn] >= SelectedStartDate && 'HR_Recruitment tracker'[Modified chuẩn] <= SelectedEndDate) ||
                ('HR_Recruitment tracker'[Created chuẩn] >= SelectedStartDate && 'HR_Recruitment tracker'[Modified chuẩn] <= SelectedEndDate)
            )
        )
    )
)
3.UV Unsatisfy= 
VAR CurrentLvl = SELECTEDVALUE('Bảng mã tuyển dụng'[Mã])

RETURN
IF(
    ISBLANK(CurrentLvl), 
    BLANK(),
    VAR SelectedStartDate = MIN('Table_date'[Date])
    VAR SelectedEndDate = MAX('Table_date'[Date])
    
    RETURN
    CALCULATE(
        COUNT('HR_Recruitment tracker'[Id]),
        'HR_Recruitment tracker'[Result] = "Unsatisfy",
        FILTER(
            ALLSELECTED('HR_Recruitment tracker'), -- Thay ALL bằng ALLSELECTED
            'HR_Recruitment tracker'[Mã (L0-L5)] = CurrentLvl && (
                ('HR_Recruitment tracker'[Created chuẩn] >= SelectedStartDate && 'HR_Recruitment tracker'[Created chuẩn] <= SelectedEndDate) ||
                ('HR_Recruitment tracker'[Modified chuẩn] >= SelectedStartDate && 'HR_Recruitment tracker'[Modified chuẩn] <= SelectedEndDate) ||
                ('HR_Recruitment tracker'[Created chuẩn] >= SelectedStartDate && 'HR_Recruitment tracker'[Modified chuẩn] <= SelectedEndDate)
            )
        )
    )
)
4.TỔNG ỨNG VIÊN = 
VAR SelectedStartDate = MIN('Table_date'[Date])
VAR SelectedEndDate = MAX('Table_date'[Date])

RETURN

    
    CALCULATE(COUNTROWS(
        FILTER('HR_Recruitment tracker',(
        ('HR_Recruitment tracker'[Created chuẩn] >= SelectedStartDate
        &&
            'HR_Recruitment tracker'[Created chuẩn]<= SelectedEndDate
     )|| 
     ('HR_Recruitment tracker'[Modified chuẩn] >= SelectedStartDate
        &&
            'HR_Recruitment tracker'[Modified chuẩn]<= SelectedEndDate
     )||
     ('HR_Recruitment tracker'[Created chuẩn] >= SelectedStartDate
        &&
            'HR_Recruitment tracker'[Modified chuẩn]<= SelectedEndDate
     )
)))
,CROSSFILTER('HR_Recruitment tracker'[Created chuẩn],Table_date[Date],None)
)
